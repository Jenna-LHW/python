```
def findSum(digit):
    sum = 0
    
    for i in digit:
        i=int(i)
        sum+=i
        
    return sum

if __name__=="__main__":
    digit=input("Enter digit: ")
    ans=findSum(digit)
    print(f"The sum of its digits are: {ans}")
```
