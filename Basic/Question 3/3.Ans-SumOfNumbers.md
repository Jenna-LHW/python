```
def findSum(n):
    sum=0
    for i in range(n+1):
        print(f"{i} + " if i<n else f"{i} = ", end="")
        sum+=i
    print(sum)   

if __name__ == "__main__":
    n = int(input("Enter a number: "))
    
    findSum(n)
```
